        """Matchbox net encoder model

        Args:
            activation (str, optional): Activation function for the layers of the network. Defaults to "relu".
            feat_in (int, optional): number of features inputed. Defaults to 64.
            normalization_mode (str, optional): Normalization mode for the normalization layers in the matchbox submodule. Defaults to "batch".
            residual_mode (str, optional): Residual mode for the residual conncetions of the network. Defaults to "add".
            norm_groups (int, optional): Defaults to -1.
            conv_mask (bool, optional): Defaults to True.
            frame_splicing (int, optional): Defaults to 1.
            init_mode (Optional[str], optional): Weight initialisation strategy. Defaults to 'xavier_uniform'.
        """
        super(ConvASREncoder, self).__init__()
        activation = conv_activations[activation]()

[docs]    def forward(self, audio_signal, length=None):
        if len(audio_signal.shape)==3:
            s_input, length = self.encoder(([audio_signal.permute(0,2,1)], length))
        else:
            raise TypeError("Unsupported input type : input must be of shape [batch,time,mfcc_features]")
        if length is None:
            return s_input[-1]

        return s_input[-1], length
